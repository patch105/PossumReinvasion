Time to equilibrium (Appendix S2: Table S1)

Calculate the mean time to population equilibrium on the peninsula for each scenario. We defined an equilibrium population as being within 2% of the year-40 population (if complete re-population occurred).

#Generate the final empty dataframes
pest_perc <- NULL

# Create the function
dens_func <- function(x){
  
  #Create a unique output directory for the scenario
  output_dir <- dir.create(paste0(main_wd,"Figures", x), showWarnings = FALSE)
  
  # Initiate an empty data frame
  equilibrium <- NULL
  
  # Loop over replicates
  for (rep in seq(0,nb_replication-1)){ 
    
    # Set path to output file
    filename_nbpest <- paste0(main_wd, x, '\\2.monthly_stat_rep_',rep,'.gz')
    
    # Read file
    monthly_stat <- read.table(filename_nbpest)
    
    pest_density <- NULL
    
    # Extract possum abundance per replicate, convert to density (dividing by area of Otago Peninsula)
    pest_density <- cbind(pest_density, monthly_stat[,1]/9782.834) %>% 
      as.data.frame()
    
    # Summarise monthly density across replicates to mean density per year
    pest_density_perc <- pest_density %>% 
      mutate(year = rep(1:40, each = 12)) %>% #create a year column
      group_by(year) %>% 
      dplyr::summarise(mean = mean(V1)) 
    
    # Calculate the percentage of the year-40 density
    pest_density_perc <- pest_density_perc %>% 
      mutate(diff_final_mean = ((mean - as.numeric(pest_density_perc[40, 2])) / as.numeric(pest_density_perc[40, 2]))*100)
    
    # Filter just the years within 2% of the final 40-year abundance 
    pest_density_perc <- pest_density_perc %>% 
      filter(diff_final_mean < 2) %>% 
      filter(diff_final_mean > -2)
    
    # Then take the first year where it reaches equilibrium
    equilibrium <- rbind(equilibrium, pest_density_perc[1,1] )
    
  }
  
  equilibrium <- as.data.frame(equilibrium) %>% 
    mutate(scenario = as.factor(x), rep = row_number())
  
  pest_perc <<- dplyr::bind_rows(pest_perc, equilibrium)
  
}

dens_func_plotting <- function(x){
  
  # Initiate an empty dataframe
  annual_pest_density <- NULL

  for (rep in seq(0, nb_replication-1)){

    #Read file
    monthly_stat <- read.table(filename_nbpest) %>% .[,1]
    
    # Turn into annual stat
    annual_pest_stat <- (unname(tapply(monthly_stat, (seq_along(monthly_stat)-1) %/% 12, mean)))
    
    # Convert to possum density
    annual_pest_density <- cbind(annual_pest_density, (annual_pest_stat/9782.834)) # Set abundance as density by dividing by area of Otago Peninsula
    
  }
  
  annual_pest_density <- as.data.frame(annual_pest_density) %>% 
    mutate(scenario = as.factor(x), year = row_number())
  
  annual_pest_density_FINAL <<- dplyr::bind_rows(annual_pest_density_FINAL, annual_pest_density)
  
}

# Initiate an empty data frame
pest_mort_prop <- NULL

# Loop over replicates
for (rep in seq(0,nb_replication-1)){
  
  # Set path to output file
  filename_nbpest <- paste0(main_wd, "Worst_case_23", '\\2.monthly_stat_rep_',rep,'.gz')
  
  # Read abundance, birth, and mortality files
  annual_stat_abun <- read.table(filename_nbpest) %>% 
    .[, 1]
  
  annual_stat_abun<- (unname(tapply(annual_stat_abun, (seq_along(annual_stat_abun)-1) %/% 12, mean))) 
  
  annual_stat_mort <- read.table(filename_nbpest) %>% 
    .[, 4]
  
  annual_stat_mort<- (unname(tapply(annual_stat_mort, (seq_along(annual_stat_mort)-1) %/% 12, mean)))
  
  annual_stat_birth <- read.table(filename_nbpest) %>% 
    .[, 3]
  
  annual_stat_birth<- (unname(tapply(annual_stat_birth, (seq_along(annual_stat_birth)-1) %/% 12, mean)))
  
  # Combine into one dataframe
  annual_stat <- cbind(annual_stat_abun, annual_stat_mort, annual_stat_birth) %>% as.data.frame()

  # Calculate dispersal mortality divided over number of births
  annual_stat <- annual_stat %>%   
    mutate(prop = (annual_stat_mort/annual_stat_birth)*100 ) %>%  
     replace(is.na(.), 0) #replace NAs with 0

  # Calculate the proportion of mortality relative to population density
  if (rep == 0) {
    pest_mort_prop <- cbind(pest_mort_prop, (annual_stat[,1]/9782.834)) # Set abundance as density
    pest_mort_prop <- cbind(pest_mort_prop, annual_stat[,4])} # Mortality proportion
  
  else {
    pest_mort_prop <- rbind(pest_mort_prop, cbind((annual_stat[,1]/9782.834), annual_stat[,4]))
    
    }
}

 pest_final <- as.data.frame(pest_mort_prop) %>% mutate(Scenario = "Incomplete elimination")

# Set up the function to calculate the mean density for each time step and scenario
mean_den_ascii_func <- function(result_subf) {
  
  # Create output directory per scenario
  output_dir<-dir.create(paste0(main_wd,"Figures",result_subf,"\\OUTPUTASCII"))
  
  for (timestep in time_sequence){
    
  # Loop over all replicates
  for (rep in seq(0,nb_replication-1)){ 
    
    # Set path 
    filename_nbpest <- paste0(main_wd, result_subf, '\\distribution_rep_',rep,'\\distribution_overTime_',timestep,'_rep_',rep,'.gz')
    
    # Read abundance file, convert to density (dividing by area of Otago Peninsula)
    density <- (read.table(filename_nbpest))/9782.834 
      
    # At the first replicate, initiate a data frame. Otherwise add the density of possums in each grid cell.
    if (rep == 0) {pest_density_temp <- density} else {
      pest_density_temp <- pest_density_temp  + density}}
    
  # Calculate the mean density of possums per grid cell (averaged over replicates)
  mean_density <- pest_density_temp / nb_replication 
  
  # Rename columns to be numbers
  colnames(mean_density) <- ncol(mean_density):1
  
  # Transpose so the x axis replaces the y and vice versa
  transpose <- t(mean_density)
  
  # Reshape into a matrix, and then dataframe
  reshape_data  <- melt( as.matrix(transpose) )
  
  # Reshape again to a dataframe which will just have 3 columns
  reshape_data <- as.data.frame(reshape_data)

  colnames(reshape_data) <- c('x', 'y', 'density')

  # Load the template ascii to have projected grid coordinates
  template_ascii <- raster(paste0(main_wd,"template_raster")) 
  
  # Convert to a dataframe
  template_ascii_df <- cbind(coordinates(template_ascii), as.data.frame(template_ascii))
  
  # Reshape to also include the column 'density' from the mean_density dataframe
  template_ascii_df$density <- reshape_data$density
  
  # Subset to have three columns
  final_df <- subset(template_ascii_df, select = c("x", "y", "density"))
  #raterize
  density_rast <- rasterFromXYZ(final_df)

  #writeRaster(density_rast,filename = paste0(main_wd, "Figures", result_subf, "\\OUTPUTASCII\\meandensity_ascii_time_", timestep,".asc"), format='ascii', overwrite = T)

  }
}
